│   └── /stream/{streamId}       # 直播观看页面

├── /account                     # 账户功能（无交易功能）

├── /download                    # 下载页面
│   ├── /app                      # 移动应用下载（iOS/Android）
│   └── /desktop                  # 桌面应用下载（Windows/macOS）

├── /help                        # 帮助中心
│   ├── /faq                      # 常见问题
│   ├── /getting-started          # 入门指南

│
├── /about                       # 关于我们
│   ├── /company                  # 公司信息
│   ├── /careers                  # 招聘信息
│   ├── /contact                  # 联系我们
│   └── /press                    # 媒体报道
│
├── /blog                        # 官方博客
│   ├── /categories               # 博客分类（新闻、分析、公告等）
│   ├── /author/{authorName}      # 按作者展示文章
